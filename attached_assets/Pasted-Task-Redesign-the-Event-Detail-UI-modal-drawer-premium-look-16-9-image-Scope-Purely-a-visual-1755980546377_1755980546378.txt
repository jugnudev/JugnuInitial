import { Dialog, DialogContent, DialogClose } from "@/components/ui/dialog";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { CalendarDays, Clock, MapPin, ExternalLink, Share2, X } from "lucide-react";
import { motion } from "framer-motion";

type Props = {
  open: boolean;
  onOpenChange: (v: boolean) => void;
  event: {
    title: string;
    category?: string;
    imageUrl?: string;
    start: Date;
    end: Date;
    venueName?: string;
    city?: string;
    address?: string;
    mapsUrl?: string;
    ticketsUrl?: string;
  };
  onShare: () => void;
  onAddToGoogle: () => void;
  onDownloadIcs: () => void;
};

export default function EventModal({ open, onOpenChange, event, onShare, onAddToGoogle, onDownloadIcs }: Props) {
  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="max-w-6xl p-0 overflow-hidden bg-neutral-900/90 backdrop-blur-xl border border-white/10">
        <motion.div
          initial={{ opacity: 0, y: 8 }}
          animate={{ opacity: 1, y: 0 }}
          transition={{ duration: 0.22, ease: "easeOut" }}
          className="grid lg:grid-cols-12 gap-0"
        >
          {/* Media */}
          <div className="lg:col-span-6">
            <div className="relative w-full bg-black/50">
              <div className="relative pb-[56.25%]">
                <img
                  src={event.imageUrl}
                  alt={event.title}
                  className="absolute inset-0 h-full w-full object-cover"
                  loading="lazy"
                  decoding="async"
                />
              </div>
            </div>
          </div>

          {/* Details */}
          <div className="lg:col-span-6 flex flex-col">
            <div className="p-6 sm:p-8 flex-1">
              <div className="flex items-start justify-between gap-3">
                <h1 id="event-title" className="text-3xl sm:text-4xl font-extrabold tracking-tight">{event.title}</h1>
                <DialogClose asChild>
                  <button aria-label="Close" className="rounded-full p-2 hover:bg-white/10">
                    <X className="h-5 w-5" />
                  </button>
                </DialogClose>
              </div>

              {event.category && (
                <div className="mt-3">
                  <Badge variant="secondary" className="bg-white/10 text-white">{event.category}</Badge>
                </div>
              )}

              <div className="mt-6 space-y-4 text-sm leading-6 text-white/90">
                <div className="flex items-start gap-3">
                  <CalendarDays className="h-5 w-5 opacity-80 mt-0.5" />
                  <div>
                    <div className="font-medium">
                      {/* format date range with existing util */}
                    </div>
                    <div className="flex items-center gap-2 opacity-80">
                      <Clock className="h-4 w-4" />
                      {/* format time range */}
                    </div>
                  </div>
                </div>

                <div className="flex items-start gap-3">
                  <MapPin className="h-5 w-5 opacity-80 mt-0.5" />
                  <div>
                    <div className="font-medium">{event.venueName}</div>
                    <div className="opacity-80">{event.city}{event.address ? `, ${event.address}` : ""}</div>
                    {event.mapsUrl && (
                      <a href={event.mapsUrl} target="_blank" rel="noopener noreferrer" className="inline-flex items-center gap-1 text-amber-400 hover:underline mt-1">
                        Open in Maps <ExternalLink className="h-3.5 w-3.5" />
                      </a>
                    )}
                  </div>
                </div>
              </div>

              <div className="mt-8 border-t border-white/10 pt-4 text-sm">
                <span className="opacity-80">Don’t see your event? </span>
                <a href="/submit" className="text-amber-400 hover:underline">Request to have it listed</a>
              </div>
            </div>

            {/* Actions bar */}
            <div className="px-6 sm:px-8 py-5 bg-black/30 border-t border-white/10 flex flex-col sm:flex-row gap-3 sm:items-center sm:justify-between">
              <div className="flex gap-3">
                <Button variant="secondary" onClick={onShare} className="gap-2">
                  <Share2 className="h-4 w-4" /> Share
                </Button>
              </div>
              <div className="flex flex-wrap gap-3">
                <Button onClick={onAddToGoogle} variant="outline">Add to Google</Button>
                <Button onClick={onDownloadIcs} variant="outline">Download .ics</Button>
                {event.ticketsUrl && (
                  <a href={event.ticketsUrl} target="_blank" rel="noopener noreferrer">
                    <Button className="bg-amber-500 hover:bg-amber-600">Get Tickets</Button>
                  </a>
                )}
              </div>
            </div>
          </div>
        </motion.div>
      </DialogContent>
    </Dialog>
  );
}
